# NA imputation
# Fill the missing value by mode
getmode <- function(v) {
  uniqv <- unique(v)
  uniqv[which.max(tabulate(match(v, uniqv)))]
}
animals[,5]<-factor(animals[,5])
animals[,6]<-factor(animals[,6]) 
mode.data <- animals
mode.5 <- getmode(animals[,5])
mode.6 <- getmode(animals[,6])

na.rows <- is.na(mode.data[, 6])
